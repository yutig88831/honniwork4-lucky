    <script>

        // 簡化後的分類配色 (柔和奶茶色調)

        const categories = [

            { name: "天醫", color: "#e9edc9", desc: "聰明與財富的流動", numbers: ["13", "31", "68", "86", "49", "94", "27", "72"] },

            { name: "延年", color: "#dbe7e4", desc: "展現才幹與責任的高度", numbers: ["19", "91", "78", "87", "34", "43", "26", "62"] },

            { name: "生氣", color: "#f2e9e4", desc: "吸引貴人與樂觀的能量", numbers: ["14", "41", "67", "76", "39", "93", "28", "82"] },

            { name: "伏位", color: "#f5ebe0", desc: "安靜沉澱，等待合適的契機", numbers: ["11", "22", "33", "44", "66", "77", "88", "99"] },

            { name: "絕命", color: "#edede9", desc: "果斷勇敢的行動特質", numbers: ["12", "21", "69", "96", "48", "84", "37", "73"] },

            { name: "五鬼", color: "#e3d5ca", desc: "充滿變化的創意與靈感", numbers: ["18", "81", "79", "97", "36", "63", "24", "42"] },

            { name: "六煞", color: "#d5bdaf", desc: "細膩的情感與人際魅力", numbers: ["16", "61", "47", "74", "38", "83", "29", "92"] },

            { name: "禍害", color: "#e6beae", desc: "言語的影響力與表達能耐", numbers: ["17", "71", "89", "98", "46", "64", "23", "32"] }

        ];

 

        let allOptions = [];

        categories.forEach(cat => {

            cat.numbers.forEach(num => {

                allOptions.push({ num, cat: cat.name, color: cat.color, desc: cat.desc });

            });

        });

 

        const canvas = document.getElementById('wheelCanvas');

        const ctx = canvas.getContext('2d');

        const container = document.getElementById('wheel-container');

        const spinBtn = document.getElementById('spinBtn');

        const resultBox = document.getElementById('resultBox');

 

        let isSpinning = false;

        let currentRotation = 0;

 

        function drawWheel() {

            const size = container.offsetWidth;

            canvas.width = size;

            canvas.height = size;

            const centerX = size / 2;

            const centerY = size / 2;

            const radius = size / 2;

            const sliceAngle = (2 * Math.PI) / allOptions.length;

 

            ctx.clearRect(0, 0, size, size);

 

            allOptions.forEach((option, i) => {

                const angle = i * sliceAngle;

               

                // 畫扇形

                ctx.beginPath();

                ctx.moveTo(centerX, centerY);

                ctx.arc(centerX, centerY, radius, angle, angle + sliceAngle);

                ctx.fillStyle = option.color;

                ctx.fill();

               

                // 極簡線條邊框

                ctx.strokeStyle = "rgba(255,255,255,0.5)";

                ctx.lineWidth = 1;

                ctx.stroke();

 

                // 畫數字

                ctx.save();

                ctx.translate(centerX, centerY);

                ctx.rotate(angle + sliceAngle / 2);

                ctx.textAlign = "right";

                ctx.fillStyle = "#7f5539";

                ctx.font = `300 ${size/45}px "Noto Sans TC"`;

                ctx.fillText(option.num, radius - 20, 5);

                ctx.restore();

            });

        }

 

        function spin() {

            if (isSpinning) return;

            isSpinning = true;

           

            resultBox.style.opacity = 0;

           

            const extraDegrees = Math.floor(Math.random() * 360);

            const totalSpin = (360 * 10) + extraDegrees;

            currentRotation += totalSpin;

           

            container.style.transform = `rotate(-${currentRotation}deg)`;

 

            setTimeout(() => {

                isSpinning = false;

                calculateResult();

            }, 6000);

        }

 

        function calculateResult() {

            const rotationInside360 = currentRotation % 360;

            const sliceSize = 360 / allOptions.length;

           

            // 指針在正上方 (270度方向)

            let index = Math.floor(((rotationInside360 + 270) % 360) / sliceSize);

           

            const winner = allOptions[index];

 

            document.getElementById('resultNumber').innerText = winner.num;

            document.getElementById('resultCategory').innerText = winner.cat;

            document.getElementById('resultDesc').innerText = winner.desc;

           

            resultBox.classList.remove('hidden');

            resultBox.style.opacity = 1;

        }

 

        window.addEventListener('load', drawWheel);

        window.addEventListener('resize', drawWheel);

        spinBtn.addEventListener('click', spin);
